$$
\begin{align*}
f(x) = x^2 \cdot ln(3x) \\
f'(x) = 2x \cdot ln(3x) + x^2 \cdot \frac{1}{x} = 2x \cdot ln(3x) + x \\
f''(x) = 2 \cdot ln(3x) + 2x \cdot \frac{1}{x} + 1
\end{align*}
$$

$f(x) = 0$:

$x_1=0$

$f'(x) = 0$

$$
\begin{align*}
2x \cdot ln(3x) + x = 0 \\
x(2 \cdot ln(3x) + 1) = 0 \\
x_1 = 0 \\
2 \cdot ln(2x) + 1 = 0 \\
2 ln(3x) = -1 \\
ln(3x) = -\frac{1}{2} \\
3x = e^{-\frac{1}{2}} \\
x_2 = \frac{1}{2 \cdot \sqrt{e}}
\end{align*}
$$

\begin{align*}

f(\frac{1}{e\sqrt{e}}) = (\frac{1}{3\sqrt{e}})^2 \cdot ln(\frac{3}{3\sqrt{e}}) \\
